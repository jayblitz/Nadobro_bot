```python
import os
import asyncio
from dotenv import load_dotenv
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, CallbackContext

# Nado SDK imports (exact from official quickstart)
from nado_protocol.client import create_nado_client, NadoClientMode
from nado_protocol.utils.math import to_x18, from_x18
from nado_protocol.utils.order import build_appendix
from nado_protocol.utils.expiration import OrderType, get_expiration_timestamp
from nado_protocol.utils.bytes32 import subaccount_to_hex
from nado_protocol.utils.nonce import gen_order_nonce
from nado_protocol.engine_client.types.execute import PlaceOrderParams, OrderParams

load_dotenv()

TELEGRAM_TOKEN = os.getenv("TELEGRAM_TOKEN")
PRIVATE_KEY = os.getenv("PRIVATE_KEY")

# Create Nado client (Testnet)
client = create_nado_client(NadoClientMode.TESTNET, PRIVATE_KEY)
subaccount_hex = subaccount_to_hex(client.context.signer.address, "default")

print(f"Connected! Address: {client.context.signer.address}")

async def start(update: Update, context: CallbackContext):
    await update.message.reply_text(
        "üëã Welcome to **Nadobro** ‚Äî your Nado AI trading companion.\n\n"
        "Chat naturally or use commands:\n"
        "/positions   /balance\n"
        "/long BTC 0.01   /short ETH 0.05\n\n"
        "Try: \"What's moving on BTC?\" or \"Show my positions\""
    )

async def handle_message(update: Update, context: CallbackContext):
    text = update.message.text.lower().strip()
    reply = "Thinking... pulling live Nado data..."

    try:
        if "funding" in text or "rate" in text or "btc" in text:
            # BTC-PERP is product_id 2 (confirmed in docs)
            market_price = client.context.engine_client.get_market_price(2)
            price = from_x18(int(market_price.bid_x18))
            reply = f"üî¥ BTC-PERP live price: ${price:,.2f}\n(Funding data coming in Phase 2 ‚Äî ask again soon!)"

        elif "position" in text or "positions" in text:
            open_orders = client.context.engine_client.get_subaccount_open_orders(2, subaccount_hex)  # BTC-PERP
            if not open_orders.orders:
                reply = "No open positions right now."
            else:
                lines = ["Your open positions:"]
                for o in open_orders.orders:
                    size = from_x18(int(o.amount))
                    side = "LONG" if size > 0 else "SHORT"
                    lines.append(f"‚Ä¢ {side} {abs(size):.4f} BTC @ ${from_x18(int(o.price_x18)):,}")
                reply = "\n".join(lines)

        elif "balance" in text or "margin" in text:
            info = client.context.engine_client.get_subaccount_info(subaccount_hex)
            bal = from_x18(int(info.spot_balances[0].balance.amount)) if info.spot_balances else 0
            reply = f"Unified margin balance: ${bal:,.2f} USDT0"

        else:
            reply = "Got it! Try /positions, /balance, or a trade command like /long BTC 0.01"

    except Exception as e:
        reply = f"Data fetch error: {str(e)[:80]}\n(Make sure you deposited ‚â• $5 USDT0 on testnet)"

    await update.message.reply_text(reply)

async def long_command(update: Update, context: CallbackContext):
    args = context.args
    if len(args) < 2:
        await update.message.reply_text("Usage: /long <size>\nExample: /long 0.01")
        return

    product_id = 2  # BTC-PERP
    size = float(args[0]) if args else 0.01

    market_price = client.context.engine_client.get_market_price(product_id)
    price = from_x18(int(market_price.bid_x18)) * 1.001  # slight above market

    order = OrderParams(
        sender=subaccount_hex,
        priceX18=to_x18(price),
        amount=to_x18(size),                    # positive = long/buy
        expiration=get_expiration_timestamp(60),
        nonce=gen_order_nonce(),
        appendix=build_appendix(OrderType.DEFAULT)  # builder_id = 0 (optional)
    )

    params = PlaceOrderParams(product_id=product_id, order=order)
    result = client.market.place_order(params)

    await update.message.reply_text(
        f"‚úÖ Long {size} BTC placed on testnet!\nDigest: {result.data.digest}"
    )

async def main():
    app = Application.builder().token(TELEGRAM_TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("long", long_command))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    await app.run_polling()

if __name__ == "__main__":
    asyncio.run(main())
```
