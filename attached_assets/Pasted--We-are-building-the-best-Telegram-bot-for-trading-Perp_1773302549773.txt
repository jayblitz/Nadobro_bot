```python
from decimal import Decimal
from src.nadobro.models.database import get_user, update_user
from src.nadobro.services.nado_client import NadoClient

async def can_execute_order(user_id: int, proposed_notional: float, strategy_name: str) -> tuple[bool, str, float]:
    user = get_user(user_id)
    if not user or not user.allocated_budget:
        return False, "No budget allocated.", 0.0

    current = Decimal(str(user.current_exposure or 0))
    proposed = Decimal(str(proposed_notional))
    budget = Decimal(str(user.allocated_budget))

    if current + proposed > budget:
        return False, f"Budget limit reached (${budget:,}).", 0.0

    client = NadoClient(user.linked_signer_address, user.encrypted_linked_signer_pk)
    await client.initialize()
    health = await client.get_margin_health()

    if health < 1.35:
        return False, f"Margin health too low ({health:.2f}).", 0.0

    safe_size = min(proposed, (budget - current) * Decimal("0.95"))
    return True, "✅ Approved", float(safe_size)


async def update_exposure(user_id: int, change: float):
    user = get_user(user_id)
    new_exp = (user.current_exposure or 0) + change
    update_user(user_id, {"current_exposure": float(new_exp)})


async def emergency_flatten(user_id: int):
    user = get_user(user_id)
    client = NadoClient(user.linked_signer_address, user.encrypted_linked_signer_pk)
    await client.initialize()
    await client.close_all_positions()
    update_user(user_id, {"current_exposure": 0.0})
    print(f"🚨 EMERGENCY FLATTEN for user {user_id}")
```

```python
import asyncio
from src.nadobro.services.budget_guard import can_execute_order, update_exposure, emergency_flatten
from src.nadobro.services.howl_service import run_howl_optimization
from src.nadobro.services.nado_client import NadoClient
from src.nadobro.models.database import get_user

class AlphaAgent:
    def __init__(self, user_id: int):
        self.user_id = user_id
        self.user = get_user(user_id)
        self.client = NadoClient(self.user.linked_signer_address, self.user.encrypted_linked_signer_pk)
        self.running = False

    async def start(self, budget: float, risk_level: str = "balanced", max_positions: int = 2):
        self.running = True
        self.user.allocated_budget = budget
        update_user(self.user_id, {"allocated_budget": budget, "alpha_active": True})
        
        await self.client.initialize()
        
        while self.running:
            opportunities = await self._scan_opportunities()
            allocation = self._wolf_allocate(opportunities, budget, risk_level, max_positions)
            
            for slot in allocation:
                success = await self._execute_slot(slot)
                if not success:
                    await emergency_flatten(self.user_id)
                    break
            
            await asyncio.sleep(300)  # 5 minutes re-evaluation

    async def _scan_opportunities(self):
        # Simple scanner — expand later with volume/funding signals
        return await self.client.get_top_opportunities(limit=5)

    def _wolf_allocate(self, opps, budget, risk_level, max_pos):
        # Smart multi-slot allocation (WOLF pattern)
        risk_multiplier = {"conservative": 0.3, "balanced": 0.6, "aggressive": 1.0}[risk_level]
        slot_size = (budget * risk_multiplier) / max_pos
        return [{"strategy": opp["type"], "pair": opp["symbol"], "size": slot_size, "price": opp["price"]} for opp in opps[:max_pos]]

    async def _execute_slot(self, slot):
        proposed = slot["size"] * slot["price"]
        allowed, reason, safe_size = await can_execute_order(self.user_id, proposed, slot["strategy"])
        
        if not allowed:
            await self._send_alert(f"🛑 {reason}")
            return False
        
        slot["size"] = safe_size / slot["price"]
        success = await self.client.place_order(**slot)
        
        if success:
            await update_exposure(self.user_id, safe_size)
        return success

    async def stop(self):
        self.running = False
        update_user(self.user_id, {"alpha_active": False})
        await emergency_flatten(self.user_id)
```
