### 1. New File: `src/nadobro/services/llm_agent.py`

```python
import os
import json
from openai import AsyncOpenAI

client = AsyncOpenAI(
    api_key=os.getenv("XAI_API_KEY"),
    base_url="https://api.x.ai/v1"
)

async def llm_analyze_opportunity(market_data: dict) -> dict:
    """Grok decides the best action with reasoning"""
    prompt = f"""
    You are Nadobro's Alpha Agent. Analyze this market snapshot and decide the best move.
    Return ONLY valid JSON with these keys: action, confidence, reason, suggested_leverage.

    Market Data: {json.dumps(market_data)}

    Possible actions: grid_long, grid_short, delta_neutral, volume_farm, skip
    """

    response = await client.chat.completions.create(
        model="grok-3-mini",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.2,
        max_tokens=400
    )

    try:
        return json.loads(response.choices[0].message.content)
    except:
        return {"action": "skip", "confidence": 0, "reason": "LLM parse error"}


async def llm_optimize_howl(trade_summary: dict) -> dict:
    """HOWL nightly optimization using Grok"""
    prompt = f"""
    Analyze these last 24h trades and suggest better parameters.
    Return ONLY JSON: new_grid_spacing, new_leverage_cap, new_rebalance_threshold

    Trade Summary: {json.dumps(trade_summary)}
    """

    response = await client.chat.completions.create(
        model="grok-3-mini",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.1
    )

    try:
        return json.loads(response.choices[0].message.content)
    except:
        return {"new_grid_spacing": 0.7, "new_leverage_cap": 10, "new_rebalance_threshold": 0.5}
```

### 2. Updated `src/nadobro/services/alpha_agent.py` (full version with LLM)

```python
import asyncio
import json
from src.nadobro.services.budget_guard import can_execute_order, update_exposure, emergency_flatten
from src.nadobro.services.llm_agent import llm_analyze_opportunity, llm_optimize_howl
from src.nadobro.services.nado_client import NadoClient
from src.nadobro.models.database import get_user, update_user

class AlphaAgent:
    def __init__(self, user_id: int):
        self.user_id = user_id
        self.user = get_user(user_id)
        self.client = NadoClient(self.user.linked_signer_address, self.user.encrypted_linked_signer_pk)
        self.running = False

    async def start(self, budget: float, risk_level: str = "balanced", max_positions: int = 2):
        self.running = True
        update_user(self.user_id, {
            "allocated_budget": budget,
            "alpha_active": True,
            "risk_level": risk_level,
            "max_positions": max_positions
        })

        await self.client.initialize()
        print(f"🤖 Alpha Agent started for user {self.user_id} | Budget: ${budget}")

        while self.running:
            # LLM-Powered Scanner
            raw_opps = await self.client.get_top_opportunities(limit=6)
            decisions = []
            for opp in raw_opps:
                decision = await llm_analyze_opportunity(opp)
                if decision["action"] != "skip" and decision["confidence"] > 60:
                    decisions.append({**opp, **decision})

            # WOLF Allocation
            allocation = self._wolf_allocate(decisions, budget, risk_level, max_positions)

            for slot in allocation:
                await self._execute_slot(slot)

            await asyncio.sleep(300)  # 5 min cycle

    def _wolf_allocate(self, decisions, budget, risk_level, max_pos):
        risk_mult = {"conservative": 0.3, "balanced": 0.6, "aggressive": 1.0}[risk_level]
        slot_size = (budget * risk_mult) / max_pos
        return [{"strategy": d["action"], "pair": d["symbol"], "size": slot_size, "price": d["price"]} for d in decisions[:max_pos]]

    async def _execute_slot(self, slot):
        proposed = slot["size"] * slot["price"]
        allowed, reason, safe_size = await can_execute_order(self.user_id, proposed, slot["strategy"])
        
        if not allowed:
            await self._send_alert(f"🛑 Agent stopped: {reason}")
            await emergency_flatten(self.user_id)
            return False

        slot["size"] = safe_size / slot["price"]
        success = await self.client.place_order(**slot)
        
        if success:
            await update_exposure(self.user_id, safe_size)
        return success

    async def stop(self):
        self.running = False
        update_user(self.user_id, {"alpha_active": False})
        await emergency_flatten(self.user_id)
```

import re

async def handle_alpha_activation(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.lower()
    
    if "alpha agent" in text or "autonomous" in text or "start agent" in text:
        # Parse budget and risk level
        budget_match = re.search(r"\$?(\d+)", text)
        budget = float(budget_match.group(1)) if budget_match else 500.0
        
        risk = "balanced"
        if "aggressive" in text:
            risk = "aggressive"
        elif "conservative" in text:
            risk = "conservative"

        agent = AlphaAgent(update.effective_user.id)
        asyncio.create_task(agent.start(budget, risk))

        await update.message.reply_text(
            f"🤖 **Alpha Agent Mode ACTIVATED**\n\n"
            f"Budget: **${budget}**\n"
            f"Risk Level: **{risk.title()}**\n"
            f"Max Positions: **2**\n\n"
            "The agent is now scanning, deciding, and trading autonomously.\n"
            "Type 'stop alpha' to pause anytime."
        )
